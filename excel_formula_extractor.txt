import tkinter as tk
from tkinter import ttk, filedialog, messagebox
import json
import zipfile
import xml.etree.ElementTree as ET
from pathlib import Path
import openpyxl
from openpyxl.utils import get_column_letter
import re


class ExcelParserApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Excel外部文件提取工具")
        self.root.geometry("800x600")

        self.file_path = None
        self.workbook = None
        self.sheet_names = []
        self.selected_sheets = []
        self.password = None

        self.create_widgets()

    def create_widgets(self):
        # 文件选择区域
        file_frame = ttk.LabelFrame(self.root, text="文件选择", padding=10)
        file_frame.pack(fill="x", padx=10, pady=5)

        self.file_entry = ttk.Entry(file_frame, width=60)
        self.file_entry.pack(side="left", padx=5)

        ttk.Button(file_frame, text="选择Excel文件", command=self.select_file).pack(side="left")

        # Sheet选择区域
        sheet_frame = ttk.LabelFrame(self.root, text="Sheet选择", padding=10)
        sheet_frame.pack(fill="both", expand=True, padx=10, pady=5)

        self.sheet_listbox = tk.Listbox(sheet_frame, selectmode="extended", height=10)
        self.sheet_listbox.pack(fill="both", expand=True, padx=5, pady=5)

        sheet_btn_frame = ttk.Frame(sheet_frame)
        sheet_btn_frame.pack(pady=5)
        ttk.Button(sheet_btn_frame, text="全选", command=self.select_all_sheets).pack(side="left", padx=5)
        ttk.Button(sheet_btn_frame, text="取消全选", command=self.deselect_all_sheets).pack(side="left", padx=5)

        # 密码区域
        pwd_frame = ttk.LabelFrame(self.root, text="密码(可选-用于加密Excel)", padding=10)
        pwd_frame.pack(fill="x", padx=10, pady=5)

        self.pwd_entry = ttk.Entry(pwd_frame, show="*", width=30)
        self.pwd_entry.pack(side="left", padx=5)

        # 状态显示
        self.status_label = ttk.Label(self.root, text="状态: 就绪", foreground="blue")
        self.status_label.pack(pady=5)

        # 操作按钮
        btn_frame = ttk.Frame(self.root)
        btn_frame.pack(pady=10)

        ttk.Button(btn_frame, text="提取外部文件引用", command=self.extract_external_refs).pack(side="left", padx=5)
        ttk.Button(btn_frame, text="清空", command=self.clear_all).pack(side="left", padx=5)

        # 结果显示区域
        result_frame = ttk.LabelFrame(self.root, text="结果预览", padding=10)
        result_frame.pack(fill="both", expand=True, padx=10, pady=5)

        self.result_text = tk.Text(result_frame, height=10, width=80)
        self.result_text.pack(fill="both", expand=True)

        scrollbar = ttk.Scrollbar(self.result_text, command=self.result_text.yview)
        scrollbar.pack(side="right", fill="y")
        self.result_text.config(yscrollcommand=scrollbar.set)

    def select_file(self):
        file_path = filedialog.askopenfilename(
            filetypes=[("Excel文件", "*.xlsx *.xlsm *.xls"), ("所有文件", "*.*")]
        )
        if file_path:
            self.file_path = file_path
            self.file_entry.delete(0, tk.END)
            self.file_entry.insert(0, file_path)
            self.load_sheets()

    def load_sheets(self):
        try:
            pwd = self.pwd_entry.get() if self.pwd_entry.get() else None
            self.workbook = openpyxl.load_workbook(self.file_path, data_only=False, password=pwd)
            self.sheet_names = self.workbook.sheetnames

            self.sheet_listbox.delete(0, tk.END)
            for name in self.sheet_names:
                self.sheet_listbox.insert(tk.END, name)

            self.status_label.config(text=f"状态: 加载成功，共{len(self.sheet_names)}个Sheet", foreground="green")
        except Exception as e:
            if "password" in str(e).lower() or "encrypted" in str(e).lower():
                self.prompt_password()
            else:
                self.status_label.config(text=f"状态: 加载失败 - {str(e)}", foreground="red")
                messagebox.showerror("错误", f"加载Excel失败: {str(e)}")

    def prompt_password(self):
        pwd_win = tk.Toplevel(self.root)
        pwd_win.title("输入密码")
        pwd_win.geometry("300x120")
        pwd_win.transient(self.root)
        pwd_win.grab_set()

        ttk.Label(pwd_win, text="Excel文件已加密，请输入密码:").pack(pady=10)
        pwd_entry = ttk.Entry(pwd_win, show="*", width=30)
        pwd_entry.pack(pady=5)

        def submit():
            self.password = pwd_entry.get()
            pwd_win.destroy()
            self.load_sheets_with_password()

        ttk.Button(pwd_win, text="确定", command=submit).pack(pady=5)
        pwd_entry.focus()
        pwd_entry.bind("<Return>", lambda e: submit())

    def load_sheets_with_password(self):
        try:
            self.workbook = openpyxl.load_workbook(self.file_path, data_only=False, password=self.password)
            self.sheet_names = self.workbook.sheetnames

            self.sheet_listbox.delete(0, tk.END)
            for name in self.sheet_names:
                self.sheet_listbox.insert(tk.END, name)

            self.status_label.config(text=f"状态: 解密成功，共{len(self.sheet_names)}个Sheet", foreground="green")
        except Exception as e:
            self.status_label.config(text=f"状态: 密码错误或解密失败", foreground="red")
            messagebox.showerror("错误", "密码错误或无法解密此文件")

    def select_all_sheets(self):
        self.sheet_listbox.select_set(0, tk.END)

    def deselect_all_sheets(self):
        self.sheet_listbox.selection_clear(0, tk.END)

    def get_selected_sheets(self):
        selected = []
        for i in self.sheet_listbox.curselection():
            selected.append(self.sheet_listbox.get(i))
        return selected

    def extract_external_refs(self):
        if not self.file_path:
            messagebox.showwarning("警告", "请先选择Excel文件")
            return

        selected_sheets = self.get_selected_sheets()
        if not selected_sheets:
            messagebox.showwarning("警告", "请至少选择一个Sheet")
            return

        self.status_label.config(text="状态: 正在提取...", foreground="orange")
        self.root.update()

        results = {
            "file": self.file_path,
            "sheets_analyzed": selected_sheets,
            "external_file_refs": [],
            "formula_external_refs": []
        }

        # 正则匹配外部文件引用模式
        external_patterns = [
            r'\[([^\]]+)\]',  # [filename.xlsx]
            r"'(.*?)'!",  # 'Sheet Path'!
            r"INDIRECT\([^)]*PATH[^)]*\)",
            r"'\.\.\\([^']+)'!",
            r"'\.\.\\([^']+)'",
        ]

        try:
            for sheet_name in selected_sheets:
                if sheet_name not in self.workbook.sheetnames:
                    continue

                sheet = self.workbook[sheet_name]

                for row in sheet.iter_rows():
                    for cell in row:
                        if cell.value is None:
                            continue

                        cell_ref = f"{get_column_letter(cell.column)}{cell.row}"
                        cell_value = str(cell.value)

                        # 检查外部文件引用
                        has_external = False
                        external_refs = []

                        # 匹配 [filename.xlsx] 模式
                        file_refs = re.findall(r'\[([^\]]+\.(xlsx|xlsm|xls|xla|xlb))\]', cell_value, re.IGNORECASE)
                        for ref in file_refs:
                            has_external = True
                            external_refs.append(ref[0])

                        # 匹配 '../' 或 '..\\' 路径模式
                        path_refs = re.findall(r"['\"]?(\.\.[\\/][^\"'\)]+['\"]?)", cell_value)
                        for ref in path_refs:
                            clean_ref = ref.strip("'\"")
                            if clean_ref and clean_ref not in external_refs:
                                external_refs.append(clean_ref)

                        # 检查公式中的外部引用
                        if cell.data_type == 'f' or (cell.value and str(cell.value).startswith('=')):
                            # 公式中的外部文件引用
                            formula = cell_value if cell_value.startswith('=') else f"={cell_value}"

                            # 查找公式中的外部引用
                            formula_file_refs = re.findall(r'\[([^\]]+\.(xlsx|xlsm|xls|xla|xlb))\]', formula, re.IGNORECASE)
                            for ref in formula_file_refs:
                                has_external = True
                                if ref[0] not in external_refs:
                                    external_refs.append(ref[0])

                            # 查找公式中的路径引用
                            formula_path_refs = re.findall(r"'\.\.\\([^']+)'", formula)
                            for ref in formula_path_refs:
                                if ref not in external_refs:
                                    external_refs.append(f"..\\{ref}")

                            formula_path_refs2 = re.findall(r"'\.\.\\/([^']+)'", formula)
                            for ref in formula_path_refs2:
                                if ref not in external_refs:
                                    external_refs.append(f"../{ref}")

                            # 检查是否包含外部链接的关键字
                            if any(keyword in formula.upper() for keyword in ['PATH', 'EXTERNAL', 'REFERENCE', 'LINK']):
                                has_external = True

                            if has_external and external_refs:
                                results["formula_external_refs"].append({
                                    "sheet": sheet_name,
                                    "cell": cell_ref,
                                    "formula": formula,
                                    "external_references": external_refs
                                })

                        # 普通单元格中的外部引用
                        if has_external and external_refs and cell.data_type != 'f':
                            results["external_file_refs"].append({
                                "sheet": sheet_name,
                                "cell": cell_ref,
                                "value": cell_value,
                                "external_references": external_refs
                            })

            # 生成JSON文件
            output_path = filedialog.asksaveasfilename(
                defaultextension=".json",
                filetypes=[("JSON文件", "*.json")],
                initialfile="excel_external_refs.json"
            )

            if output_path:
                with open(output_path, 'w', encoding='utf-8') as f:
                    json.dump(results, f, ensure_ascii=False, indent=2)

                self.status_label.config(text=f"状态: 提取完成，已保存到{output_path}", foreground="green")

                # 预览结果
                self.result_text.delete(1.0, tk.END)
                preview = json.dumps(results, ensure_ascii=False, indent=2)
                self.result_text.insert(1.0, preview[:5000])  # 限制预览长度

                messagebox.showinfo("成功", f"提取完成！\n共找到 {len(results['external_file_refs'])} 个外部文件引用单元格\n共找到 {len(results['formula_external_refs'])} 个公式外部引用\n\n已保存到: {output_path}")

            self.clear_inputs()

        except Exception as e:
            self.status_label.config(text=f"状态: 提取失败 - {str(e)}", foreground="red")
            messagebox.showerror("错误", f"提取失败: {str(e)}")

    def clear_inputs(self):
        self.file_entry.delete(0, tk.END)
        self.pwd_entry.delete(0, tk.END)
        self.sheet_listbox.delete(0, tk.END)
        self.workbook = None
        self.file_path = None
        self.password = None
        self.result_text.delete(1.0, tk.END)

    def clear_all(self):
        self.clear_inputs()
        self.status_label.config(text="状态: 已清空", foreground="blue")


if __name__ == "__main__":
    root = tk.Tk()
    app = ExcelParserApp(root)
    root.mainloop()
