from spire.xls import Workbook
import re

def col_number_to_letter(col: int) -> str:
    result = ""
    while col > 0:
        col, remainder = divmod(col - 1, 26)
        result = chr(65 + remainder) + result
    return result

def extract_external_refs(file_path: str):
    workbook = Workbook()
    workbook.LoadFromFile(file_path)

    results = []

    try:
        for sheet in workbook.Worksheets:
            sheet_name = sheet.Name
            used = sheet.AllocatedRange

            if used is None or used.RowCount == 0:
                continue

            start_row = used.Row
            end_row   = start_row + used.RowCount - 1
            start_col = used.Column
            end_col   = start_col + used.ColumnCount - 1

            for r in range(start_row, end_row + 1):
                for c in range(start_col, end_col + 1):
                    cell = sheet.Range[r, c]

                    if not cell.HasFormula:
                        continue

                    formula = cell.Formula or ""

                    if "[" in formula and "]" in formula and "!" in formula:
                        files = list(set(re.findall(r"\[([^\]]+)\]", formula)))
                        if files:
                            addr = f"{col_number_to_letter(cell.Column)}{cell.Row}"

                            val = cell.Value
                            if val is None:
                                val = cell.Text or "<无法获取值>"

                            results.append({
                                "sheet": sheet_name,
                                "cell": addr,
                                "value": str(val),
                                "formula": formula,
                                "external_files": files
                            })
    finally:
        workbook.Dispose()

    return results


if __name__ == "__main__":
    excel_path = r"D:\document\test\test.xlsx"

    data = extract_external_refs(excel_path)

    print(f"共找到 {len(data)} 个引用外部文件的公式单元格:\n")
    for item in data:
        print(f"Sheet  : {item['sheet']}")
        print(f"Cell   : {item['cell']}")
        print(f"Value  : {item['value']}")
        print(f"Files  : {', '.join(item['external_files'])}")
        print(f"Formula: {item['formula']}")
        print("-" * 50)
