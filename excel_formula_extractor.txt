from spire.xls import Workbook
import re
import json
import os
import tkinter as tk
from tkinter import ttk, filedialog, messagebox
from concurrent.futures import ProcessPoolExecutor, as_completed
import multiprocessing


def col_number_to_letter(col: int) -> str:
    """将列号转换为Excel列字母"""
    result = ""
    while col > 0:
        col, remainder = divmod(col - 1, 26)
        result = chr(65 + remainder) + result
    return result


def _is_external_ref(text: str) -> tuple:
    """检测是否为外部文件引用，返回 (是否外部引用, 外部文件名列表)"""
    if not text:
        return False, []
    if "[" in text and "]" in text and "!" in text:
        files = list(set(re.findall(r"\[([^\]]+)\]", text)))
        if files:
            return True, files
    return False, []


def _has_self_sheet_ref(text: str, sheet_names: list, current_sheet: str) -> str or None:
    """检测是否引用自身工作簿的其他Sheet，返回引用的Sheet名"""
    if not text:
        return None
    for name in sheet_names:
        if name == current_sheet:
            continue
        pattern = r"(?:'?" + re.escape(name) + r"'?\s*!)"
        if re.search(pattern, text):
            return name
    return None


def _process_single_sheet(args):
    """处理单个sheet，在子进程中执行"""
    file_path, sheet_index, sheet_name = args

    workbook = Workbook()
    workbook.LoadFromFile(file_path)

    external_results = []
    self_results = []

    try:
        sheet = workbook.Worksheets[sheet_index]
        used = sheet.AllocatedRange

        if used is None or used.RowCount == 0:
            return [], []

        start_row = used.Row
        end_row = start_row + used.RowCount - 1
        start_col = used.Column
        end_col = start_col + used.ColumnCount - 1

        for r in range(start_row, end_row + 1):
            for c in range(start_col, end_col + 1):
                cell = sheet.Range[r, c]
                addr = f"{col_number_to_letter(cell.Column)}{cell.Row}"

                val = cell.Value
                if val is None:
                    val = cell.Text or ""
                val_str = str(val)

                base_info = {
                    "sheet": sheet_name,
                    "cell": addr,
                    "value": val_str,
                }

                # 只处理公式引用
                if cell.HasFormula:
                    formula = cell.Formula or ""
                    is_ext, ext_files = _is_external_ref(formula)
                    if is_ext:
                        info = dict(base_info)
                        info["formula"] = formula
                        info["ref_type"] = "公式"
                        info["external_files"] = ext_files
                        external_results.append(info)

    finally:
        workbook.Dispose()

    return external_results, self_results


def extract_all_refs(file_path: str, use_multiprocessing: bool = True):
    """提取所有公式引用，支持多进程优化"""

    # 首先获取所有sheet信息
    workbook = Workbook()
    workbook.LoadFromFile(file_path)

    sheets_info = []
    sheet_names = []
    total_rows = 0

    try:
        for i, sheet in enumerate(workbook.Worksheets):
            sheets_info.append((file_path, i, sheet.Name))
            sheet_names.append(sheet.Name)
            used = sheet.AllocatedRange
            if used:
                total_rows += used.RowCount or 0
    finally:
        workbook.Dispose()

    # 判断是否使用多进程：至少3个sheet 且 总行数超过5000
    should_use_mp = (use_multiprocessing and
                     len(sheets_info) >= 3 and
                     total_rows >= 5000)

    external_results = []
    self_results = []

    # 单进程模式
    if not should_use_mp:
        for args in sheets_info:
            ext, self_refs = _process_single_sheet(args)
            external_results.extend(ext)
            self_results.extend(self_refs)
    else:
        # 多进程模式
        num_workers = min(len(sheets_info), multiprocessing.cpu_count())

        with ProcessPoolExecutor(max_workers=num_workers) as executor:
            futures = {executor.submit(_process_single_sheet, args): args for args in sheets_info}

            for future in as_completed(futures):
                try:
                    ext, self_refs = future.result()
                    external_results.extend(ext)
                    self_results.extend(self_refs)
                except Exception as e:
                    print(f"处理sheet时出错: {e}")

    # 处理自身Sheet引用（需要sheet_names信息）
    for item in self_results:
        if item.get("ref_type") == "公式":
            formula = item.get("formula", "")
            ref_sheet = _has_self_sheet_ref(formula, sheet_names, item["sheet"])
            if ref_sheet:
                item["ref_sheet"] = ref_sheet

    # 过滤有效的自身引用
    self_results = [item for item in self_results if item.get("ref_sheet")]

    return external_results, self_results


def export_to_json(external_results, self_results, output_path):
    """导出结果到JSON文件，只输出sheet、cell和value"""
    # 简化外部引用数据
    simple_external = []
    for item in external_results:
        simple_external.append({
            "sheet": item.get("sheet", ""),
            "cell": item.get("cell", ""),
            "value": item.get("value", ""),
        })

    # 简化自身引用数据
    simple_self = []
    for item in self_results:
        simple_self.append({
            "sheet": item.get("sheet", ""),
            "cell": item.get("cell", ""),
            "value": item.get("value", ""),
        })

    data = {
        "external_refs": simple_external,
        "self_sheet_refs": simple_self,
        "summary": {
            "external_count": len(external_results),
            "self_count": len(self_results),
        }
    }
    with open(output_path, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
    return output_path


class ExcelRefAnalyzerGUI:
    def __init__(self, root):
        self.root = root
        self.root.title("Excel 公式引用分析工具")
        self.root.geometry("1000x750")
        self.root.minsize(900, 650)

        self.external_results = []
        self.self_results = []

        # 文件选择区域
        file_frame = ttk.Frame(root, padding=10)
        file_frame.pack(fill=tk.X)

        self.file_path_var = tk.StringVar()
        ttk.Label(file_frame, text="Excel文件:").pack(side=tk.LEFT)
        ttk.Entry(file_frame, textvariable=self.file_path_var, width=50).pack(side=tk.LEFT, padx=5, fill=tk.X, expand=True)
        ttk.Button(file_frame, text="浏览...", command=self.browse_file).pack(side=tk.LEFT, padx=5)
        ttk.Button(file_frame, text="开始分析", command=self.analyze).pack(side=tk.LEFT, padx=5)
        ttk.Button(file_frame, text="导出JSON", command=self.export_json).pack(side=tk.LEFT, padx=5)

        # 选项区域
        option_frame = ttk.Frame(root, padding=5)
        option_frame.pack(fill=tk.X)

        self.use_mp_var = tk.BooleanVar(value=True)
        ttk.Checkbutton(option_frame, text="启用多进程优化（大文件自动生效）",
                       variable=self.use_mp_var).pack(side=tk.LEFT, padx=10)

        ttk.Label(option_frame, text=f"CPU核心数: {multiprocessing.cpu_count()}").pack(side=tk.LEFT, padx=10)

        # 结果显示区域
        result_frame = ttk.Frame(root, padding=10)
        result_frame.pack(fill=tk.BOTH, expand=True)

        self.notebook = ttk.Notebook(result_frame)
        self.notebook.pack(fill=tk.BOTH, expand=True)

        # 外部引用页面
        external_frame = ttk.Frame(self.notebook)
        self.notebook.add(external_frame, text="外部文件引用")

        ext_cols = ("Sheet", "Cell", "Value", "External Files", "Formula")
        self.external_tree = ttk.Treeview(external_frame, columns=ext_cols, show="headings")
        for col, width in zip(ext_cols, [100, 100, 200, 150, 300]):
            self.external_tree.heading(col, text=col)
            self.external_tree.column(col, width=width)

        ext_sb_y = ttk.Scrollbar(external_frame, orient=tk.VERTICAL, command=self.external_tree.yview)
        ext_sb_x = ttk.Scrollbar(external_frame, orient=tk.HORIZONTAL, command=self.external_tree.xview)
        self.external_tree.configure(yscrollcommand=ext_sb_y.set, xscrollcommand=ext_sb_x.set)
        self.external_tree.pack(side=tk.LEFT, fill=tk.BOTH, expand=True)
        ext_sb_y.pack(side=tk.RIGHT, fill=tk.Y)
        ext_sb_x.pack(side=tk.BOTTOM, fill=tk.X)

        # 自身引用页面
        self_frame = ttk.Frame(self.notebook)
        self.notebook.add(self_frame, text="自身Sheet引用")

        self_cols = ("Sheet", "Cell", "Value", "Ref Sheet", "Formula")
        self.self_tree = ttk.Treeview(self_frame, columns=self_cols, show="headings")
        for col, width in zip(self_cols, [100, 100, 200, 150, 300]):
            self.self_tree.heading(col, text=col)
            self.self_tree.column(col, width=width)

        self_sb_y = ttk.Scrollbar(self_frame, orient=tk.VERTICAL, command=self.self_tree.yview)
        self_sb_x = ttk.Scrollbar(self_frame, orient=tk.HORIZONTAL, command=self.self_tree.xview)
        self.self_tree.configure(yscrollcommand=self_sb_y.set, xscrollcommand=self_sb_x.set)
        self.self_tree.pack(side=tk.LEFT, fill=tk.BOTH, expand=True)
        self_sb_y.pack(side=tk.RIGHT, fill=tk.Y)
        self_sb_x.pack(side=tk.BOTTOM, fill=tk.X)

        # 状态栏
        self.status_var = tk.StringVar(value="请选择Excel文件并点击分析")
        status_bar = ttk.Label(root, textvariable=self.status_var, relief=tk.SUNKEN, anchor=tk.W)
        status_bar.pack(fill=tk.X, side=tk.BOTTOM)

    def browse_file(self):
        file_path = filedialog.askopenfilename(
            title="选择Excel文件",
            filetypes=[("Excel files", "*.xlsx *.xls *.xlsm"), ("All files", "*.*")]
        )
        if file_path:
            self.file_path_var.set(file_path)

    def analyze(self):
        file_path = self.file_path_var.get().strip()
        if not file_path:
            messagebox.showwarning("警告", "请先选择Excel文件")
            return

        # 清空之前的结果
        for item in self.external_tree.get_children():
            self.external_tree.delete(item)
        for item in self.self_tree.get_children():
            self.self_tree.delete(item)

        self.external_results = []
        self.self_results = []
        self.status_var.set("正在分析...")
        self.root.update_idletasks()

        try:
            use_mp = self.use_mp_var.get()
            ext_res, self_res = extract_all_refs(file_path, use_multiprocessing=use_mp)
            self.external_results = ext_res
            self.self_results = self_res

            # 填充外部引用表格
            for item in ext_res:
                self.external_tree.insert("", tk.END, values=(
                    item["sheet"],
                    item["cell"],
                    item.get("value", "")[:50],
                    ", ".join(item.get("external_files", [])),
                    item.get("formula", "")[:80],
                ))

            # 填充自身引用表格
            for item in self_res:
                self.self_tree.insert("", tk.END, values=(
                    item["sheet"],
                    item["cell"],
                    item.get("value", "")[:50],
                    item.get("ref_sheet", ""),
                    item.get("formula", "")[:80],
                ))

            self.status_var.set(
                f"分析完成: 外部文件引用 {len(ext_res)} 个, "
                f"自身Sheet引用 {len(self_res)} 个"
            )

            if ext_res:
                self.notebook.select(0)
            elif self_res:
                self.notebook.select(1)

        except Exception as e:
            messagebox.showerror("错误", f"分析失败: {str(e)}")
            self.status_var.set(f"分析失败: {str(e)}")

    def export_json(self):
        if not self.external_results and not self.self_results:
            messagebox.showwarning("警告", "没有可导出的结果，请先进行分析")
            return

        default_name = "excel_refs_result.json"
        if self.file_path_var.get():
            base = os.path.splitext(os.path.basename(self.file_path_var.get()))[0]
            default_name = f"{base}_refs.json"

        output_path = filedialog.asksaveasfilename(
            title="导出JSON结果",
            defaultextension=".json",
            initialfile=default_name,
            filetypes=[("JSON files", "*.json"), ("All files", "*.*")]
        )
        if not output_path:
            return

        try:
            export_to_json(self.external_results, self.self_results, output_path)
            messagebox.showinfo("成功", f"结果已导出到:\n{output_path}")
            self.status_var.set(f"已导出到: {output_path}")
        except Exception as e:
            messagebox.showerror("错误", f"导出失败: {str(e)}")


if __name__ == "__main__":
    multiprocessing.freeze_support()
    root = tk.Tk()
    app = ExcelRefAnalyzerGUI(root)
    root.mainloop()
